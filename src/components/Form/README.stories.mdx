export const Template = (args, { argTypes }) => ({
  props: Object.keys(argTypes),
  components: { Form, Checkbox, Button, TextField },
  data() {
    return {
      checkbox: false,
      textField: '',
    };
  },
  methods: {
    handleSubmit(event) {
      event.preventDefault();
      this.checkbox = false;
      this.textField = '';
      alert('Form submitted');
    },
  },
  template: `<Form @submit="handleSubmit">
    <Checkbox id="signUpCheckbox" v-model="checkbox">
      <template slot="label">Sign up for the Polaris newsletter</template>
    </Checkbox>
    <TextField v-model="textField" label="Email" type="email">
      <template slot="help-text">
        <span>
          We'll use this email address to inform you on future changes to
          Polaris.
        </span>
      </template>
    </TextField>
    <Button submit>Submit</Button>
  </Form>`,
});
